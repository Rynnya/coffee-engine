CMAKE_MINIMUM_REQUIRED(VERSION 3.11...3.25)

SET(PROJECT_NAME coffee-engine)
PROJECT(${PROJECT_NAME} CXX)

SET(BUILD_SHARED_LIBS ON)
SET(CMAKE_CXX_STANDARD 17)
SET(CMAKE_LIBRARY_OUTPUT_DIRECTORY "${CMAKE_CURRENT_SOURCE_DIR}/lib")

FILE(GLOB_RECURSE SOURCES ${CMAKE_CURRENT_SOURCE_DIR}/src/*.cpp)

FIND_PACKAGE(Vulkan REQUIRED)
ADD_SUBDIRECTORY(libs/fmt)
ADD_SUBDIRECTORY(libs/glfw)

ADD_LIBRARY(${PROJECT_NAME} SHARED ${SOURCES})
TARGET_INCLUDE_DIRECTORIES(${PROJECT_NAME} PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/include ${Vulkan_INCLUDE_DIRS})
TARGET_LINK_LIBRARIES(${PROJECT_NAME} PRIVATE Vulkan::Vulkan fmt::fmt glfw)
SET_TARGET_PROPERTIES(${PROJECT_NAME} fmt glfw PROPERTIES VS_GLOBAL_VcpkgEnabled FALSE)
